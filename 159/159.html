void  redLedBlick(void)

{
    //LED ON
    digitalWrite(RGBLEDPOWER,1);
    digitalWrite(RED, 1);
    digitalWrite(GREEN, 0);
    digitalWrite(BLUE, 0);
    
    delay(500);
    
    //LED OFF
    digitalWrite(RGBLEDPOWER,0);
    digitalWrite(RED, 0);
    digitalWrite(GREEN, 0);
    digitalWrite(BLUE, 0);
    
    delay(500);
}
